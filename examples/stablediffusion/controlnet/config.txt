#! dgenerate 1.1.0

# Use an already processed image with the canny control net

runwayml/stable-diffusion-v1-5 \
--inference-steps 40 \
--guidance-scales 8 \
--gen-seeds 2 \
--prompts "Painting, Girl with a pearl earing by Leonardo Da Vinci, masterpiece; low quality, low resolution, blank eyeballs" \
--control-nets lllyasviel/sd-controlnet-canny;scale=0.5 \
--control-images "../../media/vermeer_canny_edged.png" \
--output-path pearlearing


# Use the built in mirror and canny preprocessor action to process
# the image for you before using it for generation. This mirrors
# the image horizontally and then runs canny edge detection on it

runwayml/stable-diffusion-v1-5 \
--inference-steps 40 \
--guidance-scales 8 \
--gen-seeds 2 \
--prompts "Detailed watercolor painting of a woman sitting, facing the viewer, bright; dark" \
--control-nets lllyasviel/sd-controlnet-canny;scale=0.8 \
--control-images "../../media/monalisa.png" \
--control-image-preprocessors "mirror" "canny;lower=50;upper=100;output-file=monalisa/monalisa-canny.png" \
--output-path monalisa \
--output-size 800


# Utilize the OpenPose ControlNet with an OpenPose rigging image
# that is generated by the built in OpenPose image preprocessor action

runwayml/stable-diffusion-v1-5 \
--inference-steps 40 \
--guidance-scales 8 \
--output-path boxer \
--gen-seeds 2 \
--output-size 1024 \
--vae-tiling \
--vae-slicing \
--prompts "A boxer throwing a punch in the ring" \
--control-nets lllyasviel/sd-controlnet-openpose;scale=1.0 \
--control-images "../../media/man-fighting-pose.jpg" \
--control-image-preprocessors "openpose;include-hand=true;include-face=true;output-file=boxer/boxer-openpose.png"
